- ## systemctl使用
	- ### 配置服务
	  
	  ```bash
	  export EDITOR=vi
	  systemctl edit --full litellm
	  
	  [Unit]
	  Description=LiteLLM Proxy Server
	  After=network.target
	  
	  [Service]
	  Type=simple
	  User=root
	  WorkingDirectory=/root/project/litellm
	  ExecStart=/root/project/litellm/.venv/bin/litellm --config /root/project/litellm/config.yaml
	  Restart=always
	  RestartSec=5
	  
	  # 关键：添加 Python 库路径
	  Environment="LD_LIBRARY_PATH=/root/project/litellm/.venv/lib:/usr/lib:/usr/local/python3.12/lib"
	  Environment="PATH=/root/project/litellm/.venv/bin:/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin"
	  
	  [Install]
	  WantedBy=multi-user.target
	  ```
	- ### 启动服务
	  
	  ```shell
	  systemctl daemon-reload
	  systemctl restart litellm
	  ```
